    while(program_launched)
    {
        SDL_Event event;

        while (SDL_PollEvent(&event))
        {
            switch (event.type)
            {
                case SDL_KEYDOWN:
                    switch (event.key.keysym.sym)
                    {
                        case SDLK_b:
                            printf("apuiyer b");
                            continue;

                        default:
                            continue;;
                    }

                case SDL_KEYUP:
                    switch (event.key.keysym.sym)
                    {
                        case SDLK_b:
                            printf("relacher b");
                            continue;

                        default:
                            continue;;
                    }  


                case SDL_QUIT:
                    program_launched = SDL_FALSE;
                    break;
            
                default:
                    break;
            }
           
            
        }
        
    }
